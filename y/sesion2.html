                      controllerApp.service('Service', function() {
                        this.add = function(data, value) {
                            data.push(value)
                            return data;
                        };
                        this.borrar = function(data, id) {
                            data.splice(id, 1);
                            return data
                        };
                    })

                        var controllerApp = angular.module('ctrl', []);//Creamos el modulo
                        //Generamos el controllador llamado crlIndex
                        controllerApp.controller('ctrlIndex', function($scope,Service) {
                           $scope.user="sergio@ptree.com.mx"
                            $scope.array = [];
                            $scope.getEdad= function(){
                             $scope.obj.edad=new Date().getFullYear()-$scope.obj.date.getFullYear();       
                            }
                            $scope.addUser=function(){
                               $scope.array = Service.add($scope.array, $scope.obj);                
                              $scope.obj={};
                            }
                            $scope.deleteUser=function (index){
                              $scope.array = Service.borrar($scope.array, index);
                            }
                        })
